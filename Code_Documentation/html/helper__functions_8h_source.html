59double ang(Vertex* v1,Vertex* v2,Vertex* v3)
60{
61 // Extract the Coordinates
62 double x1 = v1->x;
63 double y1 = v1->y;
64 double x2 = v2->x;
65 double y2 = v2->y;
66 double x3 = v3->x;
67 double y3 = v3->y;
68
69 //Components of the first Vector
70 double X1 = x2-x1;
71 double Y1 = y2-y1;
72 //Components of the second Vector
73 double X2 = x2-x3;
74 double Y2 = y2-y3;
75 //The Dot Product of the Vectors
76 double dot = X1*X2+Y1*Y2;
77 //The Determinant of the vectors
78 double det = X1*Y2-X2*Y1;
79 //The Angle in Degrees is returned
80 double result = Convert(atan2(det,dot));
81 return result;
82}

145bool sameSide(Vertex *v1, Vertex*v2, Vertex*v3, Vertex*v4)
146{
147 //Slope of the line with points v1 and v2
148 double m = (v2->y - v1->y) / (v2->x - v1->x);
149 //Y intercept of the line with points v1 and v2
150 double c = v1->y - m*(v1->x);
151 //Projection of V3 onto the line
152 double p1 = v3->y - m*(v3->x) - c;
153 //Projection of V4 onto the line
154 double p2 = v4->y - m*(v4->x) - c;
155 //If Both Projections have the same sign then return true
156 if(p1*p2 >= 0) return true;
157 else return false;
158}

219 }

225 if(((v1->y > v->y) != (v2->y > v->y)))

230 count++;
231 }
232 }
233 }

236 return true;
